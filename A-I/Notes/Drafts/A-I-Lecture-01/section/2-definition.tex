\section*{Перечень используемых сокращений}
\addcontentsline{toc}{section}{Перечень используемых сокращений}
\begin{center}
\begin{tabular}{|m {15em}| m{15em}|}
    \hline
    $\mathbb{N}$ & Множество натуральных чисел\\
    \hline
    $\mathbb{Z}$ & Множество целых чисел\\
    \hline
    $\mathbb{Q}$ & Множество рациональных чисел \\
    \hline
    $\mathbb{R}$ & Множество действительных чисел \\
    \hline
    $p$ & Простое число \\
    \hline
    $a | b$ & Число $a$ делит число $b$\\
    \hline
    $[a,b]$ & НОК чисел $a$ и $b$ \\
    \hline
    $(a,b)$ & НОД чисел $a$ и $b$ \\
    \hline
    \begin{equation*}
    \begin{aligned}
        A \: & = \: \{x_{1}a_{1} + \dots + x_{n}a_{n}, \: x_{i} \in \mathbb{Z}\} \: \\ & = \: (a_{1}, \dots, a_{n})
    \end{aligned}
    \end{equation*} & $(a_{1}, \dots, a_{n})$ - Идеал\\
    \hline
    $\sigma(n)$ & Сумма делителей числа n \\
    \hline
    $\nu(n)$ & Число делителей числа n \\
    \hline
    $ord_{p}(a)$ & Порядок/показатель числа $a$ по основанию $p$ \\
    \hline
    $\forall$ & Для любого \\
    \hline
    $\exists$ & Существует \\
    \hline
    $\left[x\right]$ & Целая часть числа $x$ \\
    \hline
\end{tabular}
\end{center}
